//Assignment Problem â€“ Brute Force
#include <iostream>
using namespace std;

int n;
int cost[10][10];
int assign[10];
int best_assign[10];
int min_total = 1000000;

void swap(int &a, int &b) {
    int t = a; a = b; b = t;
}

void permute(int l, int r) {
    if(l == r) {
        int total = 0;
        for(int i=0;i<n;i++) total += cost[i][assign[i]];
        if(total < min_total) {
            min_total = total;
            for(int i=0;i<n;i++) best_assign[i] = assign[i];
        }
        return;
    }

    for(int i=l;i<=r;i++) {
        swap(assign[l], assign[i]);
        permute(l+1, r);
        swap(assign[l], assign[i]);
    }
}

int main() {
    cout << "Enter number of tasks/workers: ";
    cin >> n;
    cout << "Enter cost matrix:\n";
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin >> cost[i][j];

    for(int i=0;i<n;i++) assign[i] = i;

    permute(0, n-1);

    cout << "Minimum total cost: " << min_total << "\nAssignment:\n";
    for(int i=0;i<n;i++)
        cout << "Task " << i << " -> Worker " << best_assign[i] << "\n";

    return 0;
}
